```javascript
// Installe : npm install payplug
// Fichier : /api/payment/create-intent.js

import Payplug from 'payplug';

Payplug.setSecretKey(process.env.PAYPLUG_SECRET_KEY);

export async function createPayment(order) {
  const payment = await Payplug.Payment.create({
    amount: Math.round(order.total * 100), // en centimes
    currency: 'EUR',
    billing: {
      email: order.email,
      first_name: order.firstName,
      last_name: order.lastName,
      address1: order.address.line1,
      postcode: order.address.postcode,
      city: order.address.city,
      country: order.address.country,
    },
    hosted_payment: {
      return_url: `${process.env.NEXT_PUBLIC_URL}/order/confirmation?orderId=${order.id}`,
      cancel_url: `${process.env.NEXT_PUBLIC_URL}/cart`,
    },
    notification_url: `${process.env.NEXT_PUBLIC_URL}/api/payment/webhook`,
    metadata: {
      order_id: order.id,
    },
  });
  return payment;
}
```
